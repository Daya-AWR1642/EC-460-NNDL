import pandas as pd
# List of Tuples
data1 = [('Jack',34,'Sydney',5),
        ('Riti', 31,'Delhi',7),
        ('Aadi',46,'New York',11)]

# List of Tuples
data2 = [('Mohit',34,'Tokyo',11),
        ('Veena',31,'London',10),
        ('Shaun',36,'Las Vegas',12)]

# List of Tuples
data3 = [('Mark',47,'Mumbai',13),
        ('Jose',43,'Yokohama',14),
        ('Ramu',49,'Paris',15)]

# creating the Dataframe  object  from list of tuples
firstDf = pd.DataFrame(data1,
                      columns = ['Name','age','place','experience'],

Application of loc and iloc functions¶

  • when it comes to selecting rows and columns of the pandas DataFrame, loc and iloc are two commonly used function
  • here is the subtle difference between these two functions
  • loc selects row and columns with sepecific labels
  • iloc selects the row and column at the specific integer postions
